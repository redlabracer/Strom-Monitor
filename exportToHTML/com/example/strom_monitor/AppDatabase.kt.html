3    import android.content.Context
4    import androidx.room.Database
5    import androidx.room.Room
6    import androidx.room.RoomDatabase
7    import androidx.sqlite.db.SupportSQLiteDatabase
8    import kotlinx.coroutines.CoroutineScope
9    import kotlinx.coroutines.Dispatchers
10   import kotlinx.coroutines.launch
11   
12   @Database(entities = [Person::class], version = 1, exportSchema = false)
13   abstract class AppDatabase : RoomDatabase() {
14   
15       abstract fun personDao(): PersonDao
16   
17       companion object {
18           @Volatile
19           private var INSTANCE: AppDatabase? = null
20   
21           fun getDatabase(context: Context): AppDatabase {
22               return INSTANCE ?: synchronized(this) {
23                   val instance = Room.databaseBuilder(
24                       context.applicationContext,
25                       AppDatabase::class.java,
26                       "strom_database"
27                   )
28                       .addCallback(DatabaseCallback()) // Callback hier ohne Parameter
29                       .build()
30                   INSTANCE = instance
31                   instance
32               }
33           }
34       }
35   
36       private class DatabaseCallback : RoomDatabase.Callback() {
37           override fun onCreate(db: SupportSQLiteDatabase) {
38               super.onCreate(db)
39               INSTANCE?.let { database ->
40                   CoroutineScope(Dispatchers.IO).launch {
41                       val personDao = database.personDao()
42                       if (personDao.getCount() == 0) {
43                           val anfangsPersonen = listOf(
44                               Person(name = "Person 1", verbrauchteKwh = 0.0),
45                               Person(name = "Person 2", verbrauchteKwh = 0.0),
46                               Person(name = "Person 3", verbrauchteKwh = 0.0),
47                               Person(name = "Person 4", verbrauchteKwh = 0.0)
48                           )
49                           personDao.insertAll(anfangsPersonen)
50                       }
51                   }
52               }
53           }
54       }
55   }
