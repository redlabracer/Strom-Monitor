1    package com.example.strom_monitor
2    
3    import android.view.LayoutInflater
4    import android.view.View
5    import android.view.ViewGroup
6    import android.widget.Button
7    import android.widget.EditText
8    import android.widget.TextView
9    import android.widget.Toast
10   import androidx.recyclerview.widget.RecyclerView
11   import java.text.NumberFormat
12   import java.util.Locale
13   
14   // Wichtige Imports, die zuvor gefehlt haben könnten
15   import com.example.strom_monitor.R
16   import com.example.strom_monitor.Person
17   
18   class PersonListAdapter(
19       private val onUpdate: (Person) -> Unit
20   ) : RecyclerView.Adapter<PersonListAdapter.PersonViewHolder>() {
21   
22       private var personen = emptyList<Person>()
23   
24       companion object {
25           const val PREIS_PRO_KWH = 0.35
26           const val GRUNDGEBUEHR_MONATLICH = 10.00
27       }
28   
29       class PersonViewHolder(itemView: View) : RecyclerView.ViewHolder(itemView) {
30           val nameTextView: TextView = itemView.findViewById(R.id.textViewName)
31           val kwhTextView: TextView = itemView.findViewById(R.id.textViewKwh)
32           val kostenTextView: TextView = itemView.findViewById(R.id.textViewKosten)
33           val gesamtkostenTextView: TextView = itemView.findViewById(R.id.textViewGesamtkosten)
34           val addKwhEditText: EditText = itemView.findViewById(R.id.editTextAddKwh)
35           val addKwhButton: Button = itemView.findViewById(R.id.buttonAddKwh)
36       }
37   
38       override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): PersonViewHolder {
39           val itemView = LayoutInflater.from(parent.context)
40               .inflate(R.layout.list_item_person, parent, false)
41           return PersonViewHolder(itemView)
42       }
43   
44       override fun getItemCount() = personen.size
45   
46       override fun onBindViewHolder(holder: PersonViewHolder, position: Int) {
47           val currentPerson = personen[position]
48           val context = holder.itemView.context
49   
50           val currencyFormat = NumberFormat.getCurrencyInstance(Locale.GERMANY)
51           val numberFormat = NumberFormat.getNumberInstance(Locale.GERMANY)
52           numberFormat.maximumFractionDigits = 2
53   
54           val kostenVerbrauch = currentPerson.verbrauchteKwh * PREIS_PRO_KWH
55           val gesamtkosten = kostenVerbrauch + GRUNDGEBUEHR_MONATLICH
56   
57           holder.nameTextView.text = currentPerson.name
58           holder.kwhTextView.text = "Verbrauch: ${numberFormat.format(currentPerson.verbrauchteKwh)} kWh"
59           holder.kostenTextView.text = "Kosten Verbrauch: ${currencyFormat.format(kostenVerbrauch)}"
60           holder.gesamtkostenTextView.text = "Gesamtkosten: ${currencyFormat.format(gesamtkosten)}"
61   
62           holder.addKwhButton.setOnClickListener {
63               val kwhToAddText = holder.addKwhEditText.text.toString()
64               if (kwhToAddText.isNotEmpty()) {
65                   try {
66                       val kwhToAdd = kwhToAddText.toDouble()
67                       currentPerson.verbrauchteKwh += kwhToAdd
68                       onUpdate(currentPerson)
69                       holder.addKwhEditText.text.clear()
70                   } catch (e: NumberFormatException) {
71                       Toast.makeText(context, "Bitte eine gültige Zahl eingeben", Toast.LENGTH_SHORT).show()
72                   }
73               } else {
74                   Toast.makeText(context, "Bitte einen Wert eingeben", Toast.LENGTH_SHORT).show()
75               }
76           }
77       }
78   
79       fun setData(neuePersonen: List<Person>) {
80           this.personen = neuePersonen
81           notifyDataSetChanged()
82       }
83   }
